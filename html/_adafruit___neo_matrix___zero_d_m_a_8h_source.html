36 // These define the layout for a single 'unified' matrix (e.g. one made
37 // from NeoPixel strips, or a single NeoPixel shield), or for the pixels
38 // within each matrix of a tiled display (e.g. multiple NeoPixel shields).

40 #define NEO_MATRIX_TOP 0x00 // Pixel 0 is at top of matrix
41 #define NEO_MATRIX_BOTTOM 0x01 // Pixel 0 is at bottom of matrix
42 #define NEO_MATRIX_LEFT 0x00 // Pixel 0 is at left of matrix
43 #define NEO_MATRIX_RIGHT 0x02 // Pixel 0 is at right of matrix
44 #define NEO_MATRIX_CORNER 0x03 // Bitmask for pixel 0 matrix corner
45 #define NEO_MATRIX_ROWS 0x00 // Matrix is row major (horizontal)
46 #define NEO_MATRIX_COLUMNS 0x04 // Matrix is column major (vertical)
47 #define NEO_MATRIX_AXIS 0x04 // Bitmask for row/column layout
48 #define NEO_MATRIX_PROGRESSIVE 0x00 // Same pixel order across each line
49 #define NEO_MATRIX_ZIGZAG 0x08 // Pixel order reverses between lines
50 #define NEO_MATRIX_SEQUENCE 0x08 // Bitmask for pixel line order
51 
52 // These apply only to tiled displays (multiple matrices):
53 
54 #define NEO_TILE_TOP 0x00 // First tile is at top of matrix
55 #define NEO_TILE_BOTTOM 0x10 // First tile is at bottom of matrix
56 #define NEO_TILE_LEFT 0x00 // First tile is at left of matrix
57 #define NEO_TILE_RIGHT 0x20 // First tile is at right of matrix
58 #define NEO_TILE_CORNER 0x30 // Bitmask for first tile corner
59 #define NEO_TILE_ROWS 0x00 // Tiles ordered in rows
60 #define NEO_TILE_COLUMNS 0x40 // Tiles ordered in columns
61 #define NEO_TILE_AXIS 0x40 // Bitmask for tile H/V orientation
62 #define NEO_TILE_PROGRESSIVE 0x00 // Same tile order across each line
63 #define NEO_TILE_ZIGZAG 0x80 // Tile order reverses between lines
